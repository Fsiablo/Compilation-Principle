// break，continue跳转
int read();
void write(int);
main()
{
    int a;
    while (1)
    {
        write("请输入选择(1:break,2:continue,3:exit):\n");
        a = read();
        if (a == 1 || a == 2)
        {
            do
            {
                write("循环内\n");
                if (a == 1)
                {
                    break;
                }
                if (a == 2)
                {
                    a = a - 1;
                    continue;
                }
            } while (1);
            write("循环外\n");
        }
        else
        {
        break;
        }
    }
}
