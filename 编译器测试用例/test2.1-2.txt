//while和if嵌套测试，求最大值、最小值和平均值
main()
{
	int a=0;
	int max = 0;
	int min = 0;
	int sum = 0;
	int i = 0;
	while(i<5)
	{
		a = read();
		if(i==0){
			max = a;
			min = a;
			}
		else
		{
			if(a>max){max=a;}
			if(a<min){min=a;}
		}
		sum = sum+a;
		i = i+1;
	}
	write('\n');
	write(max);
	write('\n');
	write(min);
	write('\n');
	sum = sum/5;
	write(sum);
	write('\n');
}
